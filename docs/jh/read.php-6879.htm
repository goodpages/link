var GRID_SIZE = 10;
 
//Physics constants
var PHYS_GRAVITY = -0.1;
var PHYS_DRAG = 0.01;
var SPRING_STRENGTH = 0.2;
var SPRING_DAMPING = 0.1;
 
var HelloWorldLayer = cc.Layer.extend({
    sprite:null,
    ctor:function () {
        //////////////////////////////
        
        this._super();
        var size = cc.winSize;
        
        this.s = cc.MeshSprite.create(GRID_SIZE,GRID_SIZE, res.HelloWorld_png);
        this.addChild(this.s);
  
        var w = this.s.getContentSize().width;
        var h = this.s.getContentSize().height;
 
        for(var i = 0; i &lt; GRID_SIZE; i++)
        {
            var rowStep = i / (GRID_SIZE-1);
 
            for(var j = 0; j &lt;GRID_SIZE; j++ )
            {
                var columnStep = j / (GRID_SIZE-1);
 
                this.s.setVertice(i, j, cc.p(w*columnStep, h*(1-rowStep)));
                this.s.setTexCoord(i, j, cc.p(1*columnStep, 1*(rowStep)));
            }
        }
        
        this.physics = new ParticleSystem(PHYS_GRAVITY, PHYS_DRAG);
        
        // Init grid of particles
        this.particles = new Array(GRID_SIZE);
        for (var i = 0; i &lt; this.particles.length; i++) this.particles[i] = new Array(GRID_SIZE);
        var gridStepX = ((size.width / 2) / GRID_SIZE);
        var gridStepY = ((size.height / 2) / GRID_SIZE);
 
        // Make particles
        for (var i = 0; i &lt; GRID_SIZE; i++) {
            for (var j = 0; j &lt; GRID_SIZE; j++) {                 this.particles[i][j] = this.physics.makeParticle(0.2, j * gridStepX + (size.width / 4), (size.height - i * gridStepY) - 20, 0.0);                 if (j &gt; 0) {
                    this.physics.makeSpring(this.particles[i][j - 1], this.particles[i][j], SPRING_STRENGTH, SPRING_DAMPING, gridStepX);
                }
            }
        }
 
        // Make springs between particles
        for (var j = 0; j &lt; GRID_SIZE; j++) {
            for (var i = 1; i &lt; GRID_SIZE; i++) {
                this.physics.makeSpring(this.particles[i - 1][j], this.particles[i][j], SPRING_STRENGTH, SPRING_DAMPING, gridStepY);
            }
        }
 
        // Make the top left and top right particles fixed (not affected by forces)
        this.particles[0][0].makeFixed();
        this.particles[0][GRID_SIZE - 1].makeFixed();
        
        // Touch events
        cc.eventManager.addListener({
            event: cc.EventListener.TOUCH_ALL_AT_ONCE,
            swallowTouches: true,
            onTouchesBegan: this.onTouchesBegan,
            onTouchesMoved: this.onTouchesMoved,
            onTouchesEnded: this.onTouchesEnded
        }, this);
        
        this.scheduleUpdate();
        
        this.mp = new Array();
        
        this.mp[0] = this.particles[0][0];
        this.mp[1] = this.particles[0][GRID_SIZE-1];
        this.mp[2] = this.particles[GRID_SIZE-1][GRID_SIZE-1];
        this.mp[3] = this.particles[GRID_SIZE-1][0];
        
        return true;
    },
    onTouchesBegan:function(touches, event){
        return true;
    },
    onTouchesMoved:function(touches, event){
        var target = event.getCurrentTarget();
        
        for(var i=0; i&lt;touches.length; i++){
            if(i==4){
                break;
            }
 
            var pos = touches[i].getLocation();
            
            target.mp[i].position.set(pos.x, pos.y, 0);
            target.mp[i].velocity.clear();
        }
        
    },
    onTouchesEnded:function(touches, event){
    },
    update:function(){
        this.physics.tick();
        
        for (var j = 0; j &lt; GRID_SIZE; j++) {
            for (var i = 0; i &lt; GRID_SIZE; i++) {
                var pos = this.particles[i][j].position;
                this.s.setVertice(i, j, cc.p(pos.x, pos.y));
            }
        }
    }
});
