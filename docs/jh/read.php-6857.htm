(Note: this article first appeared as a Tip Of The Day in Flipcode. The C++ to HTML formating comes from Kurt's internal tools)



You have a bunch of rectangular pieces. You need to arrange them in a rectangular surface so that they don't overlap, keeping the total area of the rectangle as small as possible. This is fairly common when arranging characters in a bitmapped font, lightmaps for a 3D engine, and I guess other situations as well. 

The idea of this algorithm is that, as we add rectangles, we can pre-select "interesting" places where we can try to add the next rectangles. For optimal results, the rectangles should be added in order. I initially tried using area as a sorting criteria, but it didn't work well with very tall or very flat rectangles. I then tried using the longest dimension as a selector, and it worked much better. So much for intuition... 

These "interesting" places are just to the right and just below the currently added rectangle. The first rectangle, obviously, goes at the top left, the next one would go either to the right or below this one,and so on. It is a weird way to do it, but it seems to work very nicely. 

The way we search here is fairly brute-force, the fact being that for most off-line purposes the performance seems more than adequate. I have generated a japanese font with around 8500 characters and all the time was spent generating the bitmaps. 

Also, for all we care, we could grow the parent rectangle in a different way than power of two. It just happens that power of 2 is very convenient for graphics hardware textures. 

I'd be interested in hearing of other approaches to this problem. Make sure to post them on http://www.flipcode.com.

  125. bool CRectPlacement::AddAtEmptySpot   (TRect &r)  
  126. {  
  127.   // Find a valid spot among available anchors.  
  128.   
  129.   bool bFound = false;  
  130.   CPosArray::iterator it;  
  131.   for (it = m_vPositions.begin();  
  132.        !bFound && it != m_vPositions.end();  
  133.        ++it)  
  134.   {  
  135.     TRect Rect(it->x, it->y, r.w, r.h);  
  136.   
  137.     if (IsFree(Rect))  
  138.     {  
  139.       r = Rect;  
  140.       bFound = true;  
  141.       break; // Don't let the loop increase the iterator.  
  142.     }  
  143.   }  
  144.   if (bFound)  
  145.   {  
  146.     // Remove the used anchor point  
  147.     m_vPositions.erase(it);  
  148.   
  149.     // Sometimes, anchors end up displaced from the optimal position  
  150.     // due to irregular sizes of the subrects.  
  151.     // So, try to adjut it up & left as much as possible.  
  152.     for (int x = 1; x <= r.x; x++)  
  153.       if (!IsFree(TRect(r.x - x, r.y, r.w, r.h)))  
  154.         break;  
  155.     for (int y = 1; y <= r.y; y++)  
  156.       if (!IsFree(TRect(r.x, r.y - y, r.w, r.h)))  
  157.         break;  
  158.     if (y > x)  
  159.       r.y -= y-1;  
  160.     else  
  161.   
  162.       r.x -= x-1;  
  163.     AddRect(r);  
  164.   }  
  165.   return bFound;  
  166. }

  169. // --------------------------------------------------------------------------------  
  170. // Name        : AddAtEmptySpotAutoGrow  
  171. // Description : Add a rectangle of the given size, growing our area if needed  
  172. //               Area grows only until the max given.  
  173. //               Returns the placement of the rect in the rect's x,y coords  
  174. // --------------------------------------------------------------------------------  
  175.   
  176. bool CRectPlacement::AddAtEmptySpotAutoGrow   (TRect *pRect, int maxW, int maxH)  
  177. {  
  178.   if (pRect->w <= 0)  
  179.     return true;  
  180.   
  181.   int orgW = m_size.w;  
  182.   int orgH = m_size.h;  
  183.   
  184.   // Try to add it in the existing space  
  185.   while (!AddAtEmptySpot(*pRect))  
  186.   {  
  187.     int pw = m_size.w;  
  188.     int ph = m_size.h;  
  189.   
  190.     // Sanity check - if area is complete.  
  191.     if (pw >= maxW && ph >= maxH)  
  192.     {  
  193.       m_size.w = orgW;  
  194.       m_size.h = orgH;  
  195.       return false;  
  196.     }  
  197.   
  198.     // Try growing the smallest dim  
  199.     if (pw < maxW && (pw < ph || ((pw == ph) && (pRect->w >= pRect->h))))  
  200.       m_size.w = pw*2;  
  201.     else  
  202.       m_size.h = ph*2;  
  203.     if (AddAtEmptySpot(*pRect))  
  204.       break;  
  205.   
  206.     // Try growing the other dim instead  
  207.     if (pw != m_size.w)  
  208.     {  
  209.       m_size.w = pw;  
  210.       if (ph < maxW)  
  211.         m_size.h = ph*2;  
  212.     }  
  213.     else  
  214.     {  
  215.       m_size.h = ph;  
  216.       if (pw < maxW)  
  217.         m_size.w = pw*2;  
  218.     }  
  219.   
  220.     if (pw != m_size.w || ph != m_size.h)  
  221.       if (AddAtEmptySpot(*pRect))  
  222.         break;  
  223.   
  224.     // Grow both if possible, and reloop.  
  225.     m_size.w = pw;  
  226.     m_size.h = ph;  
  227.     if (pw < maxW)  
  228.       m_size.w = pw*2;  
  229.     if (ph < maxH)  
  230.       m_size.h = ph*2;  
  231.   }  
  232.   return true;  
  233. }

  70.        ++it)  
  71.   {  
  72.     // We make sure we leave one pixel between subrects, so texels don't bleed with bilinear.  
  73.     CRectPlacement::TRect r(0, 0, it->w+1, it->h+1);  
  74.   
  75.     // If an existing texture has actual space  
  76.     bool bPlaced = false;  
  77.     for (int i = 0; !bPlaced && i < vecTextures.size(); i++)  
  78.     {  
  79.       bPlaced = vecTextures[i].AddAtEmptySpotAutoGrow(&r, maxTexW, maxTexH);  
  80.       if (bPlaced)  
  81.         it->nTex = i;  
  82.     }  
  83.   
  84.     // Try starting a new texture and fit the rect in there  
  85.     if (!bPlaced)  
  86.     {  
  87.       vecTextures.push_back(CRectPlacement());  
  88.       bPlaced = vecTextures[vecTextures.size()-1].AddAtEmptySpotAutoGrow(&r, maxTexW, maxTexH);  
  89.       if (bPlaced)  
  90.         it->nTex = vecTextures.size()-1;  
  91.       else  
  92.         printf("ERROR: Subrect is too big to fit in texture!", it->w, it->h);  
  93.     }  
  94.   
  95.     // If correctly placed in a texture, the coords are returned in r.x and r.y  
  96.     // Store them.  
  97.     if (bPlaced)  
  98.     {  
  99.       it->x = r.x;  
  100.       it->y = r.y;  
  101.     }  
  102.   }  
  103. }  
  104.   
  105. // --------------------------------------------------------------------------------  
  106. // Name        : main  
  107. // Description : heh  
  108. // --------------------------------------------------------------------------------  
  109. int main()  
  110. {  
  111.   CTextureArray vecTextures;  
  112.   CSubRectArray vecSubRects;  
  113.   
  114.   CreateLetters(vecSubRects);  
  115.   CreateTextures(vecTextures, vecSubRects, MAX_TEXTURE_W, MAX_TEXTURE_H);  
  116.   
  117.   {  
  118.     for (CSubRectArray::const_iterator it = vecSubRects.begin();  
  119.          it != vecSubRects.end();  
  120.          ++it)  
  121.     {  
  122.       printf("Subrect %d (originally %d), size %dx%d, goes into texture %d at pos %d,%d\n",  
  123.              it - vecSubRects.begin(), it->n, it->w, it->h, it->nTex, it->x, it->y);  
  124.     }  
  125.   }  
  126.   
  127.   printf("Created a total of %d textures\n", vecTextures.size());  
  128.   {  
  129.     for (CTextureArray::const_iterator it = vecTextures.begin();  
  130.          it != vecTextures.end();  
  131.          ++it)  
  132.     {  
  133.       printf("  Texture %d, size %dx%d, Coverage %d / %d (%d%%)\n",  
  134.              it - vecTextures.begin(), it->GetW(), it->GetH(),  
  135.              it->GetArea(), it->GetTotalArea(), it->GetArea()*100/it->GetTotalArea());  
  136.     }  
  137.   }  
  138.   return 0;  
  139. }
