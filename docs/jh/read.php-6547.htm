{

                             };

                           };


    int i = 0;

    double x_new, y_new;

    double angleSum = 0.0;


    for(i = 0; i < 15; i++)

    {

        if(y > 0)

        {

            x_new = x + y * tangent[i];

            y_new = y - x * tangent[i];

            x = x_new;

            y = y_new;

            angleSum += angle[i];

            x_new = x - y * tangent[i];

            y_new = y + x * tangent[i];

            x = x_new;

            y = y_new;

            angleSum -= angle[i];

    }
