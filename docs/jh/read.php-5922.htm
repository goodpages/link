void SIFT::buildGaussianPyramid( const Mat& base, vector& pyr, int nOctaves ) const
{
vector sig(nOctaveLayers + 3);
pyr.resize(nOctaves*(nOctaveLayers + 3));  //pyrä¿å­˜æ‰€æœ‰ç»„æ‰€æœ‰å±‚

// precompute Gaussian sigmas using the following formula:
//  \sigma_{total}^2 = \sigma_{i}^2 + \sigma_{i-1}^2
//è®¡ç®—ç¬¬0ç»„æ¯å±‚çš„å°ºåº¦å› å­sig[i]ï¼Œç¬¬0ç»„ç¬¬0å±‚å·²ç»æ¨¡ç³Šè¿‡ï¼Œæ‰€æœ‰åªæœ‰5å±‚éœ€è¦æ¨¡ç³Š
sig[0] = sigma;                //ç¬¬0å±‚å°ºåº¦ä¸ºsigma
//fout<<"æ¯å±‚çš„å°ºåº¦ä¸ºï¼š\n";
//fout<<sig[0]<<'\n';
double k = pow( 2., 1. / nOctaveLayers );

{
double sig_prev = pow(k, (double)(i-1))*sigma;
double sig_total = sig_prev*k;
sig[i] = std::sqrt(sig_total*sig_total - sig_prev*sig_prev);
// fout<<sig[i]<<'\n';
}


//512å¤§å°çš„å›¾åƒï¼ŒnOctaves=7;
for( int o = 0; o < nOctaves; o++ )
{

//ç¬¬0ç»„ç¬¬0å±‚ä¸ºbaseå±‚ï¼Œå³åŸå§‹å›¾åƒ

// base of new octave is halved image from end of previous octave
//é«˜æ–¯é‡‘å­—å¡”çš„æ–°ç»„(new octave)çš„ç¬¬0å¹…ä¸ºä¸Šä¸€ç»„çš„ç¬¬nOctaveLayerså¹…ä¸‹é‡‡æ ·å¾—åˆ°ï¼Œé‡‡æ ·æ­¥é•¿ä¸º2
else if( i == 0 )

resize(src, dst, Size(src.cols/2, src.rows/2),

}

// æ¯ä¸€ç»„çš„ç¬¬iå¹…å›¾åƒæ˜¯ç”±è¯¥ç»„ç¬¬i-1å¹…å›¾åƒç”¨sig[i]é«˜æ–¯æ¨¡ç³Šå¾—åˆ°ï¼Œç›¸å½“äºä½¿ç”¨äº†æ–°çš„å°ºåº¦ã€‚

}

// Computes a gradient orientation histogram at a specified pixel
static float calcOrientationHist( const Mat& img, Point pt, int radius,
float sigma, float* hist, int n )
{
//lenè¡¨ç¤ºä»¥2*radius+1ä¸ºåŠå¾„çš„åœ†(å› ä¸ºç‚¹æ˜¯ç¦»æ•£çš„ï¼Œå…¶å®ä¸ºæ­£æ–¹å½¢)çš„åƒç´ ä¸ªæ•°
int i, j, k, len = (radius*2+1)*(radius*2+1);

float expf_scale = -1.f/(2.f * sigma * sigma);
AutoBuffer buf(len*4 + n+4);
float *X = buf, *Y = X + len, *Mag = X, *Ori = Y + len, *W = Ori + len;
float* temphist = W + len + 2;

for( i = 0; i < n; i++ )
temphist[i] = 0.f;

for( i = -radius, k = 0; i <= radius; i++ )
{
int y = pt.y + i;
if( y <= 0 || y >= img.rows - 1 )
continue;
for( j = -radius; j <= radius; j++ )
{
int x = pt.x + j;
if( x <= 0 || x >= img.cols - 1 )
continue;

float dx = (float)(img.at(y, x+1) - img.at(y, x-1));
float dy = (float)(img.at(y-1, x) - img.at(y+1, x));

X[k] = dx; Y[k] = dy; W[k] = (i*i + j*j)*expf_scale;
k++;
}
}

len = k;

// compute gradient values, orientations and the weights over the pixel neighborhood
exp(W, W, len);
fastAtan2(Y, X, Ori, len, true);   //è®¡ç®—å‘é‡è§’åº¦
magnitude(X, Y, Mag, len);        //è®¡ç®—æ¢¯åº¦

for( k = 0; k < len; k++ )
{
int bin = cvRound((n/360.f)*Ori[k]);
if( bin >= n )
bin -= n;
if( bin < 0 )
bin += n;
temphist[bin] += W[k]*Mag[k];
}

// smooth the histogram
//å¹³æ»‘
temphist[-1] = temphist[n-1];
temphist[-2] = temphist[n-2];
temphist[n] = temphist[0];
temphist[n+1] = temphist[1];
for( i = 0; i < n; i++ )
{
hist[i] = (temphist[i-2] + temphist[i+2])*(1.f/16.f) +
(temphist[i-1] + temphist[i+1])*(4.f/16.f) +
temphist[i]*(6.f/16.f);
}

float maxval = hist[0];
for( i = 1; i < n; i++ )
maxval = std::max(maxval, hist[i]);

return maxval;     //è¿”å›ç›´æ–¹å›¾ä¸­æœ€å¤§å€¼
}

static bool adjustLocalExtrema( const vector& dog_pyr, KeyPoint& kpt, int octv,
int& layer, int& r, int& c, int nOctaveLayers,
float contrastThreshold, float edgeThreshold, float sigma )
{
//è¡¨ç¤ºç°åº¦å½’ä¸€åŒ–è¦ç”¨åˆ°çš„å‚æ•°
const float img_scale = 1.f/(255*SIFT_FIXPT_SCALE);   //é‡‘å­—å¡”å›¾åƒç°åº¦æ‹‰ä¼¸äº†48å€ï¼Œæ‰€ä»¥è¦é™¤48.SIFT_FIXPT_SCALE=48
const float deriv_scale = img_scale*0.5f;     //ä¸€é˜¶å¯¼æ•°ç°åº¦å½’ä¸€åŒ–å‚æ•°
const float second_deriv_scale = img_scale;    //dxxï¼Œdyyã€‚äºŒé˜¶å¯¼æ•°ç°åº¦å½’ä¸€åŒ–å‚æ•°
const float cross_deriv_scale = img_scale*0.25f;    //dxyï¼Œäº¤å‰å¯¼æ•°ç°åº¦å½’ä¸€åŒ–å‚æ•°

float xi=0, xr=0, xc=0, contr;
int i = 0;

//å¾ªç¯5æ¬¡ï¼Œå¯¹åæ ‡è¿›è¡Œ5æ¬¡ä¿®æ­£ã€‚
//SIFT_MAX_INTERP_STEPS=5ï¼Œè¡¨ç¤º maximum steps of keypoint interpolation before failure
for( ; i < SIFT_MAX_INTERP_STEPS; i++ )
{
int idx = octv*(nOctaveLayers+2) + layer;  //å¾—åˆ°è¯¥ç‰¹å¾ç‚¹æ‰€åœ¨çš„dogå±‚åºå·
const Mat& img = dog_pyr[idx];
const Mat& prev = dog_pyr[idx-1];
const Mat& next = dog_pyr[idx+1];

//è®¡ç®—ä¸€é˜¶åå¯¼æ•°ï¼Œå¹¶å½’ä¸€åŒ–ï¼Œé€šè¿‡ä¸´è¿‘ç‚¹å·®åˆ†æ±‚å¾—
Vec3f dD((img.at(r, c+1) - img.at(r, c-1))*deriv_scale,
(img.at(r+1, c) - img.at(r-1, c))*deriv_scale,
(next.at(r, c) - prev.at(r, c))*deriv_scale);

//è®¡ç®—äºŒé˜¶åå¯¼æ•°ï¼Œå¹¶å½’ä¸€åŒ–ï¼Œé€šè¿‡ä¸´è¿‘ç‚¹å·®åˆ†æ±‚å¾—
float v2 = (float)img.at(r, c)*2;
float dxx = (img.at(r, c+1) + img.at(r, c-1) - v2)*second_deriv_scale;
float dyy = (img.at(r+1, c) + img.at(r-1, c) - v2)*second_deriv_scale;
float dss = (next.at(r, c) + prev.at(r, c) - v2)*second_deriv_scale;
float dxy = (img.at(r+1, c+1) - img.at(r+1, c-1) -
img.at(r-1, c+1) + img.at(r-1, c-1))*cross_deriv_scale;
float dxs = (next.at(r, c+1) - next.at(r, c-1) -
prev.at(r, c+1) + prev.at(r, c-1))*cross_deriv_scale;
float dys = (next.at(r+1, c) - next.at(r-1, c) -
prev.at(r+1, c) + prev.at(r-1, c))*cross_deriv_scale;

//äºŒé˜¶åå¯¼æ•°çŸ©é˜µ
Matx33f H(dxx, dxy, dxs,
dxy, dyy, dys,
dxs, dys, dss);

//ä»¤å°ºåº¦æ–¹ç¨‹çš„æ³°å‹’å±•å¼€å¯¼æ•°ä¸º0ï¼Œæ±‚è§£å‡ºåç§»é‡X
Vec3f X = H.solve(dD, DECOMP_LU);

xi = -X[2];      //å±‚åç§»ï¼Œå±‚åç§»å³å°ºåº¦åç§»
xr = -X[1];     //è¡Œåç§»
xc = -X[0];    //åˆ—åç§»

//å¦‚æœæ±‚è§£å‡ºçš„åç§»é‡å‡å°äº0.5ï¼Œåˆ™é€€å‡ºå¾ªç¯ï¼Œè¯´æ˜è¯¥å…³é”®ç‚¹çš„é€‰å–æ˜¯æ­£ç¡®çš„
if( std::abs( xi ) < 0.5f  &&  std::abs( xr ) < 0.5f  &&  std::abs( xc ) < 0.5f )
break;

//æ±‚è§£å‡ºçš„åç§»é‡å‡å¤§äº0.5ï¼Œåˆ™å°†åŸåæ ‡åŠ ä¸Šæ±‚å‡ºæ¥çš„åç§»é‡ï¼Œå¾—åˆ°æ›´ç²¾ç¡®çš„åæ ‡å’Œå°ºåº¦
c += cvRound( xc );
r += cvRound( xr );
layer += cvRound( xi );    

//å±‚æ•°æˆ–è¡Œåˆ—è¶…å‡ºè¾¹ç•Œåˆ™é€€å‡º
if( layer < 1 || layer > nOctaveLayers ||
c < SIFT_IMG_BORDER || c >= img.cols - SIFT_IMG_BORDER  ||
r < SIFT_IMG_BORDER || r >= img.rows - SIFT_IMG_BORDER )

}

//ä¿è¯æ’å€¼æ”¶æ•›ï¼ŒSIFT_MAX_INTERP_STEPS=5
if( i >= SIFT_MAX_INTERP_STEPS )
return false;

{

//ä»¥ä¸‹ä»£ç ç”¨æ¥å»é™¤ä½å¯¹æ¯”åº¦çš„ä¸ç¨³å®šç‰¹å¾ç‚¹ï¼ˆç°åº¦è¦å½’ä¸€åŒ–ï¼‰
int idx = octv*(nOctaveLayers+2) + layer;  //æ±‚å‡ºä¿®æ­£åçš„å±‚åºå·
const Mat& img = dog_pyr[idx];
const Mat& prev = dog_pyr[idx-1];
const Mat& next = dog_pyr[idx+1];
//æ±‚å‡ºä¿®æ­£åçš„å…³é”®ç‚¹åæ ‡çš„ä¸€é˜¶å¾®åˆ†å‘é‡
Matx31f dD((img.at(r, c+1) - img.at(r, c-1))*deriv_scale,
(img.at(r+1, c) - img.at(r-1, c))*deriv_scale,
(next.at(r, c) - prev.at(r, c))*deriv_scale);
//ä¸€é˜¶åå¯¼å‘é‡ä¸ä¸Šé¢æ±‚å‡ºçš„åç§»é‡å‘é‡æ±‚ç‚¹ç§¯
float t = dD.dot(Matx31f(xc, xr, xi));
//ä¿®æ­£åçš„åæ ‡å¸¦å…¥æ³°å‹’å±•å¼€å¼ï¼Œå¾—åˆ°çš„ç»“æœå°äºcontrastThreshold=0.04åˆ™æŠ›å¼ƒè¯¥ç‚¹
contr = img.at(r, c)*img_scale + t * 0.5f;    
if( std::abs( contr ) * nOctaveLayers < contrastThreshold )
return false;

//åˆ©ç”¨HessiançŸ©é˜µçš„è¿¹å’Œè¡Œåˆ—å¼è®¡ç®—è¯¥å…³é”®ç‚¹çš„ä¸»æ›²ç‡çš„æ¯”å€¼
float v2 = img.at(r, c)*2.f;
float dxx = (img.at(r, c+1) + img.at(r, c-1) - v2)*second_deriv_scale;
float dyy = (img.at(r+1, c) + img.at(r-1, c) - v2)*second_deriv_scale;
float dxy = (img.at(r+1, c+1) - img.at(r+1, c-1) -
img.at(r-1, c+1) + img.at(r-1, c-1)) * cross_deriv_scale;
float tr = dxx + dyy;
float det = dxx * dyy - dxy * dxy;

//edgeThreshold=10,å»é™¤è¾¹ç¼˜ç‚¹
if( det <= 0 || tr*tr*edgeThreshold >= (edgeThreshold + 1)*(edgeThreshold + 1)*det )
return false;
}

//ç²¾ç¡®ç‰¹å¾ç‚¹åœ¨åŸå›¾åƒä¸Šçš„ä½ç½®
kpt.pt.x = (c + xc) * (1 << octv);    //é«˜æ–¯é‡‘å­—å¡”åæ ‡æ ¹æ®ç»„æ•°æ‰©å¤§ç›¸åº”çš„å€æ•°
kpt.pt.y = (r + xr) * (1 << octv);
kpt.octave = octv + (layer << 8) + (cvRound((xi + 0.5)*255) << 16);   //ç‰¹å¾ç‚¹è¢«æ£€æµ‹å‡ºæ—¶æ‰€åœ¨çš„é‡‘å­—å¡”ç»„
kpt.size = sigma*powf(2.f, (layer + xi) / nOctaveLayers)*(1 << octv)*2;   //å…³é”®ç‚¹é‚»åŸŸç›´å¾„

return true;
}

void SIFT::findScaleSpaceExtrema( const vector& gauss_pyr, const vector& dog_pyr,
vector& keypoints ) const
{
int nOctaves = (int)gauss_pyr.size()/(nOctaveLayers + 3);    //ç»„çš„ä¸ªæ•°

// The contrast threshold used to filter out weak features in semi-uniform  
// (low-contrast) regions. The larger the threshold, the less features are produced by the detector.  
//ä½ å¯¹æ¯”åº¦çš„é˜ˆå€¼ï¼Œ contrastThresholdé»˜è®¤ä¸º0.04  
int threshold = cvFloor(0.5 * contrastThreshold / nOctaveLayers * 255 * SIFT_FIXPT_SCALE);
const int n = SIFT_ORI_HIST_BINS;  //ç›´æ–¹å›¾binçš„ä¸ªæ•°=36ï¼Œæ¯ä¸ª10åº¦
float hist[n];
KeyPoint kpt;

keypoints.clear();

for( int o = 0; o < nOctaves; o++ )
for( int i = 1; i <= nOctaveLayers; i++ )  // nOctaveLayersè¡¨ç¤ºæ¯å±‚Dogå›¾åƒä¸ªæ•°-2ï¼Œå³æœ€ç»ˆç”¨åˆ°çš„å±‚æ•°
{
int idx = o*(nOctaveLayers+2)+i;         
const Mat& img = dog_pyr[idx];           //è·å–è¯¥å±‚Dogå›¾åƒï¼Œåºå·ä»1å¼€å§‹ã€‚ç¬¬0å±‚å’Œæœ€åä¸€å±‚ä¸ç”¨
const Mat& prev = dog_pyr[idx-1];    //ä¸Šä¸€å¹…Dogå›¾åƒ
const Mat& next = dog_pyr[idx+1];   //ä¸‹ä¸€å¹…Dogå›¾åƒ
int step = (int)img.step1();
int rows = img.rows, cols = img.cols;

//SIFT_IMG_BORDER=5ï¼Œè¾¹ç•Œ5ä¸ªåƒç´ çš„è·ç¦»
for( int r = SIFT_IMG_BORDER; r < rows-SIFT_IMG_BORDER; r++)    
{
//è·å–3å¹…ç›¸é‚»çš„Dogå›¾åƒçš„è¡ŒæŒ‡é’ˆ
const short* currptr = img.ptr(r);
const short* prevptr = prev.ptr(r);
const short* nextptr = next.ptr(r);

for( int c = SIFT_IMG_BORDER; c < cols-SIFT_IMG_BORDER; c++)
{
//è·å–è¯¥å±‚å›¾åƒrè¡Œcåˆ—çš„åƒç´ å€¼
int val = currptr[c];

// find local extrema with pixel accuracy
//ä¸å‘¨å›´26ä¸ªç‚¹æ¯”è¾ƒï¼Œæå¤§æˆ–æå°å€¼åˆ™ä¸ºå±€éƒ¨æå€¼ç‚¹
if( std::abs(val) > threshold &&
((val > 0 && val >= currptr[c-1] && val >= currptr[c+1] &&
val >= currptr[c-step-1] && val >= currptr[c-step] && val >= currptr[c-step+1] &&
val >= currptr[c+step-1] && val >= currptr[c+step] && val >= currptr[c+step+1] &&
val >= nextptr[c] && val >= nextptr[c-1] && val >= nextptr[c+1] &&
val >= nextptr[c-step-1] && val >= nextptr[c-step] && val >= nextptr[c-step+1] &&
val >= nextptr[c+step-1] && val >= nextptr[c+step] && val >= nextptr[c+step+1] &&
val >= prevptr[c] && val >= prevptr[c-1] && val >= prevptr[c+1] &&
val >= prevptr[c-step-1] && val >= prevptr[c-step] && val >= prevptr[c-step+1] &&
val >= prevptr[c+step-1] && val >= prevptr[c+step] && val >= prevptr[c+step+1]) ||
(val < 0 && val <= currptr[c-1] && val <= currptr[c+1] &&
val <= currptr[c-step-1] && val <= currptr[c-step] && val <= currptr[c-step+1] &&
val <= currptr[c+step-1] && val <= currptr[c+step] && val <= currptr[c+step+1] &&
val <= nextptr[c] && val <= nextptr[c-1] && val <= nextptr[c+1] &&
val <= nextptr[c-step-1] && val <= nextptr[c-step] && val <= nextptr[c-step+1] &&
val <= nextptr[c+step-1] && val <= nextptr[c+step] && val <= nextptr[c+step+1] &&
val <= prevptr[c] && val <= prevptr[c-1] && val <= prevptr[c+1] &&
val <= prevptr[c-step-1] && val <= prevptr[c-step] && val <= prevptr[c-step+1] &&
val <= prevptr[c+step-1] && val <= prevptr[c+step] && val <= prevptr[c+step+1])))
{
//æ‰¾åˆ°æå€¼ç‚¹ä¹‹åï¼Œåœ¨Dogä¸­è°ƒæ•´å±€éƒ¨æå€¼ç‚¹
//è°ƒæ•´åçš„å…³é”®ç‚¹å…·æœ‰ä»¥ä¸‹ä¸‰ä¸ªå±æ€§
//åœ¨åŸå›¾ä¸­çš„ç²¾ç¡®åæ ‡ï¼Œ
//ç‰¹å¾ç‚¹æ‰€åœ¨çš„é«˜æ–¯é‡‘å­—å¡”ç»„ï¼Œå³æ›´ç²¾ç¡®çš„o
//é¢†åŸŸç›´å¾„
int r1 = r, c1 = c, layer = i;
if( !adjustLocalExtrema(dog_pyr, kpt, o, layer, r1, c1,
nOctaveLayers, (float)contrastThreshold,
(float)edgeThreshold, (float)sigma) )
continue;

float scl_octv = kpt.size*0.5f/(1 << o);   //è·å–è¯¥ç‰¹å¾ç‚¹çš„å°ºåº¦
//calcOrientationHistè®¡ç®—è¯¥ç‰¹å¾ç‚¹å‘¨å›´çš„æ–¹å‘ç›´æ–¹å›¾ï¼Œå¹¶è¿”å›ç›´æ–¹å›¾æœ€å¤§å€¼
//å‚æ•°oå’Œc1,r1å‡å·²ç»ç»è¿‡ç²¾ç¡®å®šä½
//æ–¹å‘ç›´æ–¹å›¾çš„è®¡ç®—æ˜¯åœ¨è¯¥ç‚¹å°ºåº¦çš„é«˜æ–¯é‡‘å­—å¡”å›¾åƒä¸­è®¡ç®—çš„ï¼Œä¸æ˜¯åœ¨Dogå›¾åƒï¼Œä¹Ÿä¸æ˜¯åœ¨åŸå›¾
float omax = calcOrientationHist(gauss_pyr[o*(nOctaveLayers+3) + layer],
Point(c1, r1),                                                 //ç‰¹å¾ç‚¹åæ ‡
cvRound(SIFT_ORI_RADIUS * scl_octv),  //ç›´æ–¹å›¾ç»Ÿè®¡åŠå¾„ï¼š3*1.5*σï¼ŒSIFT_ORI_RADIUS=3*1.5
SIFT_ORI_SIG_FCTR * scl_octv,              //ç›´æ–¹å›¾å¹³æ»‘æ‰€ç”¨åˆ°çš„å°ºåº¦ï¼ŒSIFT_ORI_SIG_FCTR=1.5f
hist, n); 
float mag_thr = (float)(omax * SIFT_ORI_PEAK_RATIO);   //è¾…æ–¹å‘ä¸º0.8*ä¸»æ–¹å‘æœ€å¤§å€¼ï¼ŒSIFT_ORI_PEAK_RATIO=0.8f
for( int j = 0; j < n; j++ )                    //n=36
{
int l = j > 0 ? j - 1 : n - 1;
int r2 = j < n-1 ? j + 1 : 0;

if( hist[j] > hist[l]  &&  hist[j] > hist[r2]  &&  hist[j] >= mag_thr )
{
float bin = j + 0.5f * (hist[l]-hist[r2]) / (hist[l] - 2*hist[j] + hist[r2]);
bin = bin < 0 ? n + bin : bin >= n ? bin - n : bin;
kpt.angle = (float)((360.f/n) * bin);    //å¾—åˆ°å…³é”®ç‚¹çš„æ–¹å‘
keypoints.push_back(kpt);     //è¿™é‡Œä¿å­˜çš„ç‰¹å¾ç‚¹å…·æœ‰ä½ç½®ï¼Œå°ºåº¦å’Œæ–¹å‘3ä¸ªä¿¡æ¯
}
}
}
}
}
}

 void SIFT::calcSIFTDescriptor( const Mat& img, Point2f ptf, float ori, float scl,
                               int d, int n, float* dst )
{
    Point pt(cvRound(ptf.x), cvRound(ptf.y));  //åæ ‡ç‚¹å–æ•´
    float cos_t = cosf(ori*(float)(CV_PI/180));  //ä½™å¼¦å€¼
    float sin_t = sinf(ori*(float)(CV_PI/180));   //æ­£å¼¦å€¼
    float bins_per_rad = n / 360.f;
    float exp_scale = -1.f/(d * d * 0.5f);
    float hist_width = SIFT_DESCR_SCL_FCTR * scl;
    int radius = cvRound(hist_width * 1.4142135623730951f * (d + 1) * 0.5f);
    cos_t /= hist_width;
    sin_t /= hist_width;

    int i, j, k, len = (radius*2+1)*(radius*2+1), histlen = (d+2)*(d+2)*(n+2);
    int rows = img.rows, cols = img.cols;

    AutoBuffer buf(len*6 + histlen);
    float *X = buf, *Y = X + len, *Mag = Y, *Ori = Mag + len, *W = Ori + len;
    float *RBin = W + len, *CBin = RBin + len, *hist = CBin + len;

//åˆå§‹åŒ–ç›´æ–¹å›¾
    for( i = 0; i < d+2; i++ )
    {
        for( j = 0; j < d+2; j++ )
            for( k = 0; k < n+2; k++ )
                hist[(i*(d+2) + j)*(n+2) + k] = 0.;
    }

    for( i = -radius, k = 0; i <= radius; i++ )
        for( j = -radius; j <= radius; j++ )
        {
           
            float c_rot = j * cos_t - i * sin_t;
            float r_rot = j * sin_t + i * cos_t;
            float rbin = r_rot + d/2 - 0.5f;
            float cbin = c_rot + d/2 - 0.5f;
            int r = pt.y + i, c = pt.x + j;

            if( rbin > -1 && rbin < d && cbin > -1 && cbin < d &&
               r > 0 && r < rows - 1 && c > 0 && c < cols - 1 )
            {
                float dx = (float)(img.at(r, c+1) - img.at(r, c-1));
                float dy = (float)(img.at(r-1, c) - img.at(r+1, c));
                X[k] = dx; Y[k] = dy; RBin[k] = rbin; CBin[k] = cbin;
                W[k] = (c_rot * c_rot + r_rot * r_rot)*exp_scale;
                k++;
            }
        }

    len = k;
    fastAtan2(Y, X, Ori, len, true);
    magnitude(X, Y, Mag, len);
    exp(W, W, len);

    for( k = 0; k < len; k++ )
    {
        float rbin = RBin[k], cbin = CBin[k];
        float obin = (Ori[k] - ori)*bins_per_rad;
        float mag = Mag[k]*W[k];

        int r0 = cvFloor( rbin );
        int c0 = cvFloor( cbin );
        int o0 = cvFloor( obin );
        rbin -= r0;
        cbin -= c0;
        obin -= o0;

        if( o0 < 0 )
            o0 += n;
        if( o0 >= n )
            o0 -= n;

        // histogram update using tri-linear interpolation
        float v_r1 = mag*rbin, v_r0 = mag - v_r1;
        float v_rc11 = v_r1*cbin, v_rc10 = v_r1 - v_rc11;
        float v_rc01 = v_r0*cbin, v_rc00 = v_r0 - v_rc01;
        float v_rco111 = v_rc11*obin, v_rco110 = v_rc11 - v_rco111;
        float v_rco101 = v_rc10*obin, v_rco100 = v_rc10 - v_rco101;
        float v_rco011 = v_rc01*obin, v_rco010 = v_rc01 - v_rco011;
        float v_rco001 = v_rc00*obin, v_rco000 = v_rc00 - v_rco001;

        int idx = ((r0+1)*(d+2) + c0+1)*(n+2) + o0;
        hist[idx] += v_rco000;
        hist[idx+1] += v_rco001;
        hist[idx+(n+2)] += v_rco010;
        hist[idx+(n+3)] += v_rco011;
        hist[idx+(d+2)*(n+2)] += v_rco100;
        hist[idx+(d+2)*(n+2)+1] += v_rco101;
        hist[idx+(d+3)*(n+2)] += v_rco110;
        hist[idx+(d+3)*(n+2)+1] += v_rco111;
    }

    // finalize histogram, since the orientation histograms are circular
    for( i = 0; i < d; i++ )
        for( j = 0; j < d; j++ )
        {
            int idx = ((i+1)*(d+2) + (j+1))*(n+2);
            hist[idx] += hist[idx+n];
            hist[idx+1] += hist[idx+n+1];
            for( k = 0; k < n; k++ )
                dst[(i*d + j)*n + k] = hist[idx+k];
        }
    // copy histogram to the descriptor,
    // apply hysteresis thresholding
    // and scale the result, so that it can be easily converted
    // to byte array
    float nrm2 = 0;
    len = d*d*n;
    for( k = 0; k < len; k++ )
        nrm2 += dst[k]*dst[k];
    float thr = std::sqrt(nrm2)*SIFT_DESCR_MAG_THR;
    for( i = 0, nrm2 = 0; i < k; i++ )
    {
        float val = std::min(dst[i], thr);
        dst[i] = val;
        nrm2 += val*val;
    }
    nrm2 = SIFT_INT_DESCR_FCTR/std::max(std::sqrt(nrm2), FLT_EPSILON);
    for( k = 0; k < len; k++ )
    {
        dst[k] = saturate_cast(dst[k]*nrm2);
    }
}

 void SIFT::calcDescriptors(const vector& gpyr, const vector& keypoints,
                            Mat& descriptors, int nOctaveLayers )
{
//SIFT_DESCR_WIDTH=4ï¼Œæè¿°ç›´æ–¹å›¾çš„å®½åº¦
//SIFT_DESCR_HIST_BINS=8
    int d = SIFT_DESCR_WIDTH, n = SIFT_DESCR_HIST_BINS;

    for( size_t i = 0; i < keypoints.size(); i++ )
    {
        KeyPoint kpt = keypoints[i];
        int octv=kpt.octave & 255, layer=(kpt.octave >> 8) & 255;   //è¯¥ç‰¹å¾ç‚¹æ‰€åœ¨çš„ç»„åºå·å’Œå±‚åºå·
        float scale = 1.f/(1 << octv);      //ç¼©æ”¾å€æ•°
        float size=kpt.size*scale;           //è¯¥ç‰¹å¾ç‚¹æ‰€åœ¨ç»„çš„å›¾åƒå°ºå¯¸
        Point2f ptf(kpt.pt.x*scale, kpt.pt.y*scale);   //è¯¥ç‰¹å¾ç‚¹åœ¨é‡‘å­—å¡”ç»„ä¸­çš„åæ ‡
        const Mat& img = gpyr[octv*(nOctaveLayers + 3) + layer];   //è¯¥ç‚¹æ‰€åœ¨çš„é‡‘å­—å¡”å›¾åƒ

        calcSIFTDescriptor(img, ptf, kpt.angle, size*0.5f, d, n, descriptors.ptr((int)i));
    }
}
